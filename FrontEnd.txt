# Crear un proyecto de react js 
$ npx create-react-app react-app

# Acceder a la carpeta de react
$ cd react-app

# Instalar el enroutador de react
$ npm install react-router-dom

# Instalar depedencias de react 
$ npm install axios
$ npm install --save react-toastify

# Correr el proyecto 
$ npm run dev 

// @see https://reactrouter.com/en/main/start/tutorial#adding-a-router
